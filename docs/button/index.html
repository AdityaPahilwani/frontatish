#Usage

// react base imports
import { Button } from '@growwSuper/core/ui'
const MyComponent = () => (
<Button
label="PRIMARY BUTTON"
type="primary"
onPress={() => console.log('Pressed')}
>
Press me
</Button>
);
export default MyComponent;
